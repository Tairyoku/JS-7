function weightMaintenance(weight) {
  if (weight < 4) return 'Пора перекусить';
  else if (weight >= 4 && weight <= 5.5 ) return 'Вес в норме';
  else return 'Пора на тренировку';
}
let weights = [1, 3, 4, 5, 10];
let recomendation;
for (let weight of weights) {
  recomendation = weightMaintenance(weight);
  console.log(recomendation);
}
